package parser

import (
        "fmt"
        "path/filepath"
        "strings"
        "sync"
)

var (
        // ErrUnsupportedFormat 不支持的文件格式
        ErrUnsupportedFormat = fmt.Errorf("unsupported config file format")

        // ErrParserNotFound 找不到对应的解析器
        ErrParserNotFound = fmt.Errorf("parser not found for file type")
)

// ParserFactory 解析器工厂
type ParserFactory struct {
        mu      sync.RWMutex
        parsers map[string]Parser // key: 文件扩展名, value: Parser实例
}

// globalFactory 全局工厂实例
var (
        globalFactory     *ParserFactory
        globalFactoryOnce sync.Once
)

// GetFactory 获取全局工厂实例（单例）
func GetFactory() *ParserFactory {
        globalFactoryOnce.Do(func() {
                globalFactory = NewParserFactory()
                // 注册内置解析器
                globalFactory.RegisterDefaultParsers()
        })
        return globalFactory
}

// NewParserFactory 创建解析器工厂
func NewParserFactory() *ParserFactory {
        return &ParserFactory{
                parsers: make(map[string]Parser),
        }
}

// Register 注册解析器
func (f *ParserFactory) Register(parser Parser) {
        f.mu.Lock()
        defer f.mu.Unlock()

        for _, ext := range parser.SupportedExtensions() {
                // 统一为小写，去掉可能的 . 前缀
                ext = strings.ToLower(strings.TrimPrefix(ext, "."))
                f.parsers[ext] = parser
        }
}

// RegisterDefaultParsers 注册默认支持的解析器
func (f *ParserFactory) RegisterDefaultParsers() {
        // 注册 YAML 解析器
        f.Register(NewYAMLParser())

        // 注册 JSON 解析器
        f.Register(NewJSONParser())

        // TODO: 未来在这里注册其他解析器
        // f.Register(NewTOMLParser())
        // f.Register(NewINIParser())
        // f.Register(NewPropertiesParser())
}

// GetParser 根据文件扩展名获取解析器
func (f *ParserFactory) GetParser(ext string) (Parser, error) {
        f.mu.RLock()
        defer f.mu.RUnlock()

        ext = strings.ToLower(strings.TrimPrefix(ext, "."))
        parser, ok := f.parsers[ext]
        if !ok {
                return nil, fmt.Errorf("%w: %s", ErrParserNotFound, ext)
        }
        return parser, nil
}

// GetParserByFilePath 根据文件路径获取解析器
func (f *ParserFactory) GetParserByFilePath(path string) (Parser, error) {
        ext := filepath.Ext(path)
        if ext == "" {
                return nil, fmt.Errorf("%w: no file extension in %s", ErrUnsupportedFormat, path)
        }
        return f.GetParser(ext)
}

// SupportedFormats 返回所有支持的格式
func (f *ParserFactory) SupportedFormats() []string {
        f.mu.RLock()
        defer f.mu.RUnlock()

        formats := make([]string, 0, len(f.parsers))
        seen := make(map[string]bool)

        for ext := range f.parsers {
                if !seen[ext] {
                        formats = append(formats, ext)
                        seen[ext] = true
                }
        }
        return formats
}

// ParseFile 便捷方法：自动选择解析器并解析文件
func (f *ParserFactory) ParseFile(path string) (*ParseResult, error) {
        parser, err := f.GetParserByFilePath(path)
        if err != nil {
                return nil, err
        }
        return parser.ParseFile(path)
}

// 包级别的便捷函数

// ParseFile 使用全局工厂解析文件
func ParseFile(path string) (*ParseResult, error) {
        return GetFactory().ParseFile(path)
}

// Register 注册自定义解析器到全局工厂
func Register(parser Parser) {
        GetFactory().Register(parser)
}

// SupportedFormats 获取所有支持的格式
func SupportedFormats() []string {
        return GetFactory().SupportedFormats()
}

// JSONParser JSON 文件解析器
type JSONParser struct {
        *BaseParser
}

// NewJSONParser 创建 JSON 解析器
func NewJSONParser() *JSONParser {
        return &JSONParser{
                BaseParser: NewBaseParser("JSON", []string{".json"}),
        }
}

// Parse 从 io.Reader 解析 JSON
func (p *JSONParser) Parse(reader io.Reader) (*ParseResult, error) {
        decoder := json.NewDecoder(reader)

        var raw interface{}
        if err := decoder.Decode(&raw); err != nil {
                return nil, fmt.Errorf("failed to decode JSON: %w", err)
        }

        // 复用 YAML 的树构建逻辑（JSON 和 YAML 的数据结构相同）
        root := buildConfigTree("root", raw)

        return &ParseResult{
                Root: root,
                Raw:  raw,
        }, nil
}

// ParseFile 从文件解析 JSON
func (p *JSONParser) ParseFile(path string) (*ParseResult, error) {
        file, err := os.Open(path)
        if err != nil {
                return nil, fmt.Errorf("failed to open file: %w", err)
        }
        defer file.Close()

        return p.Parse(file)
}
package parser

import "io"

/**
* The goal of the Parser is to uniformly convert
* configuration files of different formats
* (such as YAML/JSON/TOML, etc.) into a standard tree structure,
* which can be used for subsequent type inference and
* code generation.
 */

// Parser configuration file parser interface
// All parsers for various formats need to implement this interface
type Parser interface {
        // Parse the configuration file from io.Reader
        Parse(reader io.Reader) (*ParseResult, error)

        // ParseFile parses the configuration from a file path
        ParseFile(path string) (*ParseResult, error)

        // SupportedExtensions return the supported file extensions
        SupportedExtensions() []string

        // Name returns the parser name
        Name() string
}

// Provide general functions to avoid 
// each parser having to repeatedly implement the Name() 
// and SupportedExtensions() methods.
type BaseParser struct {
        name       string
        extensions []string
}

func NewBaseParser(name string, extensions []string) *BaseParser {
        return &BaseParser{
                name:       name,
                extensions: extensions,
        }
}

func (b *BaseParser) SupportedExtensions() []string {
        return b.extensions
}

func (b *BaseParser) Name() string {
        return b.name
}

package parser

// ValueType 配置值的类型
type ValueType int

const (
        TypeString ValueType = iota
        TypeInt
        TypeFloat
        TypeBool
        TypeArray
        TypeObject
        TypeNull
)

// ConfigValue 表示配置值的中间表示
// 支持基本类型、嵌套对象和数组
type ConfigValue struct {
        Type  ValueType
        Value any
}

// ConfigNode 表示配置树的节点（用于构建配置结构）
type ConfigNode struct {
        Key      string                 // 字段名
        Value    any                    // 值（基本类型、数组或 map）
        Type     ValueType              // 值类型
        Children map[string]*ConfigNode // 子节点（对象类型时使用）
        Items    []*ConfigNode          // 数组元素（数组类型时使用）
}

// ParseResult 解析结果
type ParseResult struct {
        Root *ConfigNode // 配置树的根节点
        Raw  any         // 原始解析的 map/slice
}

// NewConfigNode 创建配置节点
func NewConfigNode(key string) *ConfigNode {
        return &ConfigNode{
                Key:      key,
                Children: make(map[string]*ConfigNode),
                Items:    make([]*ConfigNode, 0),
        }
}

// AddChild 添加子节点（对象类型）
func (n *ConfigNode) AddChild(child *ConfigNode) {
        if n.Children == nil {
                n.Children = make(map[string]*ConfigNode)
        }
        n.Children[child.Key] = child
}

// AddItem 添加数组元素（数组类型）
func (n *ConfigNode) AddItem(item *ConfigNode) {
        n.Items = append(n.Items, item)
}

// IsObject 是否为对象类型
func (n *ConfigNode) IsObject() bool {
        return n.Type == TypeObject
}

// IsArray 是否为数组类型
func (n *ConfigNode) IsArray() bool {
        return n.Type == TypeArray
}

// IsPrimitive 是否为基本类型
func (n *ConfigNode) IsPrimitive() bool {
        return n.Type == TypeString || n.Type == TypeInt ||
                n.Type == TypeFloat || n.Type == TypeBool || n.Type == TypeNull
}

package parser

import (
        "fmt"
        "io"
        "os"

        "gopkg.in/yaml.v3"
)

// YAMLParser YAML 文件解析器
type YAMLParser struct {
        *BaseParser
}

// NewYAMLParser 创建 YAML 解析器
func NewYAMLParser() *YAMLParser {
        return &YAMLParser{
                BaseParser: NewBaseParser("YAML", []string{".yaml", ".yml"}),
        }
}

// Parse 从 io.Reader 解析 YAML
func (p *YAMLParser) Parse(reader io.Reader) (*ParseResult, error) {
        decoder := yaml.NewDecoder(reader)

        var raw interface{}
        if err := decoder.Decode(&raw); err != nil {
                return nil, fmt.Errorf("failed to decode YAML: %w", err)
        }

        // 构建配置树
        root := buildConfigTree("root", raw)

        return &ParseResult{
                Root: root,
                Raw:  raw,
        }, nil
}

// ParseFile 从文件解析 YAML
func (p *YAMLParser) ParseFile(path string) (*ParseResult, error) {
        file, err := os.Open(path)
        if err != nil {
                return nil, fmt.Errorf("failed to open file: %w", err)
        }
        defer file.Close()

        return p.Parse(file)
}

// buildConfigTree 递归构建配置树
func buildConfigTree(key string, value interface{}) *ConfigNode {
        node := NewConfigNode(key)

        switch v := value.(type) {
        case map[string]interface{}:
                // 对象类型
                node.Type = TypeObject
                node.Value = v
                for k, val := range v {
                        child := buildConfigTree(k, val)
                        node.AddChild(child)
                }

        case []interface{}:
                // 数组类型
                node.Type = TypeArray
                node.Value = v
                for i, item := range v {
                        itemNode := buildConfigTree(fmt.Sprintf("[%d]", i), item)
                        node.AddItem(itemNode)
                }

        case string:
                node.Type = TypeString
                node.Value = v

        case int, int8, int16, int32, int64:
                node.Type = TypeInt
                node.Value = v

        case float32, float64:
                node.Type = TypeFloat
                node.Value = v

        case bool:
                node.Type = TypeBool
                node.Value = v

        case nil:
                node.Type = TypeNull
                node.Value = nil

        default:
                // 未知类型，尝试转换为字符串
                node.Type = TypeString
                node.Value = fmt.Sprint(v)
        }

        return node
}
